import React from 'react'
import PropTypes from 'prop-types'
import UIHelpers from 'common/UIHelpers'

import GridList from '@material-ui/core/GridList'
import GridListTile from '@material-ui/core/GridListTile'
import GridListTileBar from '@material-ui/core/GridListTileBar'

/**
 * @summary CategoriesGridViewPresentation
 * @component
 * @version 1.0.1
 *
 * @prop {node} children
 * @prop {array} categories
 * @prop {number} [cols] defaults to 6
 * @prop {number} [cellHeight] defaults to 160
 */
function CategoriesGridViewPresentation({categories, cols, cellHeight }) {
  return (
    <GridList
      cols={cols}
      cellHeight={cellHeight}
    >
      {categories.map((category, index) => {
        return (
          <GridListTile
            onClick={()=>{
              console.log(category.href)
            }}
            key={`category${index}`}
          >
            <img src={UIHelpers.getThumbnail({imageObj: {}}, 'Small')} alt={''} />
            <GridListTileBar
              title={category.text}
              style={{ backgroundColor: 'rgba(180, 0, 0, 0.75)' }}
            />
          </GridListTile>
        )
      })}
    </GridList>
  )
}

// Proptypes
const { array, number } = PropTypes
CategoriesGridViewPresentation.propTypes = {
  categories: array.isRequired,
  cols: number,
  cellHeight: number,
}
CategoriesGridViewPresentation.defaultProps = {
  categories: [],
  cols: 6,
  cellHeight: 160,
}

export default CategoriesGridViewPresentation
